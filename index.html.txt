<script>
document.addEventListener('DOMContentLoaded', function() {
  const form = document.getElementById('decorForm');
  const fileInput = document.getElementById('imageUpload');
  const fileName = document.getElementById('fileName');
  const uploadPreview = document.getElementById('uploadPreview');
  const loading = document.getElementById('loading');
  const results = document.getElementById('results');
  const beforeImg = document.getElementById('beforeImg');
  const afterImg = document.getElementById('afterImg');
  
  // معاينة الصورة
  fileInput.addEventListener('change', function(e) {
    if (e.target.files.length > 0) {
      const file = e.target.files[0];
      fileName.textContent = file.name;
      
      const reader = new FileReader();
      reader.onload = function(event) {
        uploadPreview.src = event.target.result;
        uploadPreview.style.display = 'block';
      };
      reader.readAsDataURL(file);
    }
  });
  
  // إرسال النموذج
  form.addEventListener('submit', async function(e) {
    e.preventDefault();
    
    const file = fileInput.files[0];
    const description = document.getElementById('description').value;
    const style = document.getElementById('styleSelect').value;
    
    if (!file) {
      alert('يرجى اختيار صورة');
      return;
    }

    form.style.display = 'none';
    loading.style.display = 'block';
    results.style.display = 'none';

    const reader = new FileReader();
    reader.onload = async function(event) {
      const base64Image = event.target.result;
      beforeImg.src = base64Image;

      const prompt = `تحديث ديكور داخلي بنمط ${style} مع: ${description || 'تحديث عصري وأنيق'}. جدد الأثاث، وحسن الإضاءة، وأضف لمسات جمالية.`;

      try {
        const response = await fetch('/api/generate', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ imageBase64: base64Image, prompt })
        });

        const data = await response.json();
        if (!response.ok) throw new Error(data.error || 'فشل في الاتصال بالذكاء الاصطناعي');

        const pollStatus = async (getUrl) => {
          const res = await fetch(getUrl);
          const result = await res.json();

          if (result.status === 'succeeded') {
            afterImg.src = result.output;
            loading.style.display = 'none';
            results.style.display = 'flex';
            form.style.display = 'block';
          } else if (result.status === 'failed') {
            alert('فشل التوليد: ' + (result.error || 'خطأ غير معروف'));
            loading.style.display = 'none';
            form.style.display = 'block';
          } else {
            setTimeout(() => pollStatus(getUrl), 3000);
          }
        };

        pollStatus(data.prediction.urls.get);

      } catch (err) {
        alert('خطأ: ' + err.message);
        loading.style.display = 'none';
        form.style.display = 'block';
      }
    };
    reader.readAsDataURL(file);
  });
});
</script>
