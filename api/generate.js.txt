// api/generate.js
export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Only POST allowed' });
  }

  const { imageBase64, prompt } = req.body;

  if (!imageBase64 || !prompt) {
    return res.status(400).json({ error: 'Missing image or prompt' });
  }

  const REPLICATE_API_TOKEN = process.env.REPLICATE_API_TOKEN;
  if (!REPLICATE_API_TOKEN) {
    return res.status(500).json({ error: 'Server missing API token' });
  }

  try {
    const response = await fetch('https://api.replicate.com/v1/predictions', {
      method: 'POST',
      headers: {
        'Authorization': `Token ${REPLICATE_API_TOKEN}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        version: "ostris/flux-dev", // نموذج سريع ومجاني نسبيًا
        input: {
          prompt: `تصميم ديكور داخلي: ${prompt}. غرفة أنيقة، إضاءة طبيعية، أثاث عصري.`,
          image: imageBase64,
          aspect_ratio: "1:1",
          output_format: "jpeg",
          output_quality: 85
        }
      })
    });

    if (!response.ok) {
      const err = await response.json().catch(() => ({}));
      throw new Error(err.detail || 'Failed to create prediction');
    }

    const prediction = await response.json();
    res.status(200).json({ prediction });
  } catch (error) {
    console.error('API Error:', error);
    res.status(500).json({ error: error.message });
  }
}

export const config = {
  api: {
    bodyParser: { sizeLimit: '10mb' },
  },
};
